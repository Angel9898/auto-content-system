# GitHub Secrets required (example names)

# Model keys
GEMINI_API_KEY         # (optional) Google Generative API key / token for Gemini 3 Pro
CHATGPT_API_KEY        # (optional) OpenAI key for ChatGPT (or equivalent)
GEMINI_NANO_KEY        # (optional) hosted nano endpoint key
HF_API_KEY             # HuggingFace Inference API token (for SDXL images + TTS)

# YouTube (Google)
YOUTUBE_CLIENT_ID
YOUTUBE_CLIENT_SECRET
YOUTUBE_REFRESH_TOKEN

# Instagram / Facebook
FB_PAGE_ID
IG_USER_ID
FB_PAGE_ACCESS_TOKEN

# LinkedIn
LINKEDIN_CLIENT_ID
LINKEDIN_CLIENT_SECRET
LINKEDIN_ACCESS_TOKEN
LI_OWNER_URN          # e.g. urn:li:person:<id> or urn:li:organization:<id>

# X (Twitter)
X_API_KEY
X_API_SECRET
X_ACCESS_TOKEN
X_ACCESS_TOKEN_SECRET
X_BEARER_TOKEN

# Optional
CANVA_API_KEY         # optional: use Canva API for design templates

# Control flags
SELF_TEST             # set 'true' to run in self-test mode (no real publishes)
DRY_RUN               # set 'true' to skip publishing in scripts (additional safety)
